Step 9: Create Stripe Service
9.1 Create File
Create a new file: 
backend/src/services/stripeService.ts
9.2 Add Code
TypeScript
import Stripe from 'stripe';
const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
apiVersion: '2023-10-16',
});
export const stripeService = {
// Create checkout session
createCheckoutSession: async (
customerId: string,
priceId: string,
successUrl: string,
    cancelUrl: string
  ) => {
    const session = await stripe.checkout.sessions.create({
      customer: customerId,
      payment_method_types: ['card'],
      line_items: [
        {
          price: priceId,
          quantity: 1,
        },
      ],
      mode: 'subscription',
      success_url: successUrl,
      cancel_url: cancelUrl,
    });
    return session;
  },
  // Get subscription
  getSubscription: async (subscriptionId: string) => {
    const subscription = await stripe.subscriptions.retrieve(subscriptionId);
    return subscription;
  },
  // Cancel subscription
  cancelSubscription: async (subscriptionId: string) => {
    const subscription = await stripe.subscriptions.del(subscriptionId);
    return subscription;
  },
  // Verify webhook signature
  verifyWebhookSignature: (body: string, signature: string) => {
    try {
      const event = stripe.webhooks.constructEvent(
        body,
        signature,
        process.env.STRIPE_WEBHOOK_SECRET!
      );
      return event;
    } catch (error) {
      throw new Error('Invalid webhook signature');
    }
  },
};
Step 10: Create Stripe Routes
10.1 Create File
Create a new file: backend/src/routes/stripeRoutes.ts
10.2 Add Code
TypeScript
import express, { Request, Response } from 'express';
import { stripeService } from '../services/stripeService';
const router = express.Router();
// Create checkout session
router.post('/create-checkout-session', async (req: Request, res: Response) => 
  try {
    const { customerId, priceId, successUrl, cancelUrl } = req.body;
    const session = await stripeService.createCheckoutSession(
      customerId,
      priceId,
      successUrl,
      cancelUrl
    );
    res.json({ sessionId: session.id });
  } catch (error) {
    res.status(500).json({ error: (error as Error).message });
  }
});
// Get subscription
router.get('/subscription/:subscriptionId', async (req: Request, res: Response) 
  try {
    const subscription = await stripeService.getSubscription(
      req.params.subscriptionId
    );
    res.json(subscription);
  } catch (error) {
    res.status(500).json({ error: (error as Error).message });
  }
});
// Cancel subscription
router.post('/cancel-subscription', async (req: Request, res: Response) => {
  try {
    const { subscriptionId } = req.body;
    const subscription = await stripeService.cancelSubscription(subscriptionId)
    res.json(subscription);
  } catch (error) {
    res.status(500).json({ error: (error as Error).message });
  }
});
// Webhook endpoint
router.post('/webhook', express.raw({ type: 'application/json' }), async (req: R
  try {
    const signature = req.headers['stripe-signature'] as string;
    const event = stripeService.verifyWebhookSignature(req.body, signature);
    switch (event.type) {
      case 'checkout.session.completed':
        console.log('Checkout completed:', event.data.object);
        break;
      case 'customer.subscription.created':
        console.log('Subscription created:', event.data.object);
        break;
      case 'customer.subscription.updated':
        console.log('Subscription updated:', event.data.object);
        break;
      case 'customer.subscription.deleted':
        console.log('Subscription deleted:', event.data.object);
        break;
      case 'invoice.payment_succeeded':
        console.log('Payment succeeded:', event.data.object);
        break;
      case 'invoice.payment_failed':
        console.log('Payment failed:', event.data.object);
        break;
    }
    res.json({ received: true });
  } catch (error) {
    res.status(400).json({ error: (error as Error).message });
  }
});
export default router;

import React, { useState } from 'react';
import { View, Text, TouchableOpacity, ActivityIndicator } from 'react-native';
import { useStripe } from '@stripe/stripe-react-native';
export default function SubscriptionScreen() {
  const [loading, setLoading] = useState(false);
  const [selectedPlan, setSelectedPlan] = useState<'monthly' | 'annual' | null>
  const { initPaymentSheet, presentPaymentSheet } = useStripe();
  const handleSubscribe = async (plan: 'monthly' | 'annual') => {
    setLoading(true);
    setSelectedPlan(plan);
    try {
      // Call your backend to create checkout session
      const priceId = plan === 'monthly' 
        ? process.env.STRIPE_PRICE_ID_MONTHLY 
        : process.env.STRIPE_PRICE_ID_ANNUAL;
      const response = await fetch('/api/stripe/create-checkout-session', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          priceId,
          customerId: 'customer_id_here', // Get from user context
          successUrl: 'https://your-app.com/subscription/success',
          cancelUrl: 'https://your-app.com/subscription/cancel',
        } ),
      });
      const { sessionId } = await response.json();
      // Initialize payment sheet
      const { error } = await initPaymentSheet({
        merchantDisplayName: 'SeniorShield',
        customerId: 'customer_id_here',
        customerEphemeralKeySecret: 'key_here', // Get from backend
        paymentIntentClientSecret: sessionId,
      });
      if (error) {
        console.error('Error initializing payment sheet:', error);
        return;
      }
      // Present payment sheet
      const { error: presentError } = await presentPaymentSheet();
      if (presentError) {
        console.error('Error presenting payment sheet:', presentError);
      }
    } catch (error) {
      console.error('Error:', error);
    } finally {
      setLoading(false);
      setSelectedPlan(null);
    }
  };
  return (
    <View className="flex-1 bg-white p-6">
      <Text className="text-3xl font-bold mb-8">Choose Your Plan</Text>
      {/* Monthly Plan */}
      <TouchableOpacity
        onPress={() => handleSubscribe('monthly')}
        disabled={loading}
        className="mb-4 p-6 bg-gray-800 rounded-lg border-2 border-blue-500"
      >
        <Text className="text-xl font-bold text-white mb-2">Monthly Plan</Text>
        <Text className="text-3xl font-bold text-blue-400 mb-4">$19.99/month</Te
        <Text className="text-gray-300 mb-4">Cancel anytime</Text>
        {loading && selectedPlan === 'monthly' ? (
          <ActivityIndicator color="#3b82f6" />
        ) : (
          <Text className="text-white font-bold">Subscribe Now</Text>
        )}
      </TouchableOpacity>
      {/* Annual Plan */}
      <TouchableOpacity
        onPress={() => handleSubscribe('annual')}
        disabled={loading}
        className="mb-4 p-6 bg-gray-800 rounded-lg border-2 border-green-500"
      >
        <Text className="text-xl font-bold text-white mb-2">Annual Plan</Text>
        <Text className="text-3xl font-bold text-green-400 mb-2">$203.90/year</T
        <Text className="text-gray-300 mb-2">Save 15% vs monthly</Text>
        <Text className="text-gray-300 mb-4">Cancel anytime</Text>
        {loading && selectedPlan === 'annual' ? (
          <ActivityIndicator color="#10b981" />
        ) : (
          <Text className="text-white font-bold">Subscribe Now</Text>
        )}
      </TouchableOpacity>
    </View>
  );
}
